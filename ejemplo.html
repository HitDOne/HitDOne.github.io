; Este programa suma dos números y muestra el resultado 
.model small ; Define el modelo de memoria  
.stack 100h ; Reserva espacio para la pila  
.data ; Inicia la sección de datos  
num1 dw 10 ; Define una variable de 16 bits llamada num1 con el valor 10  
num2 dw 20 ; Define una variable de 16 bits llamada num2 con el valor 20  
resultado dw ? ; Define una variable de 16 bits llamada resultado sin inicializar  
.code ; Inicia la sección de código  
inicio: ; Define una etiqueta llamada inicio  
mov ax, @data ; Mueve la dirección de la sección de datos al registro AX  
mov ds, ax ; Mueve el valor del registro AX al registro DS  
mov ax, num1 ; Mueve el valor de la variable num1 al registro AX  
add ax, num2 ; Suma el valor de la variable num2 al registro AX  
mov resultado, ax ; Mueve el valor del registro AX a la variable resultado  
mov ah, 4ch ; Mueve el valor 4Ch al registro AH  
int 21h ; Interrumpe el programa y devuelve el control al sistema operativo  
end inicio ; Finaliza el programa y salta a la etiqueta inicio
